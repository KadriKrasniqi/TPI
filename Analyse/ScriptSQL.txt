/*------------------------------------------------------------
-- Table: t_image
------------------------------------------------------------*/
CREATE TABLE t_image(
	id_image INT IDENTITY (1,1) NOT NULL ,
	imaLien  VARCHAR (100) NOT NULL UNIQUE,
	fkmot   INT  NOT NULL ,
	CONSTRAINT prk_constraint_t_image PRIMARY KEY NONCLUSTERED (id_image)
);


/*------------------------------------------------------------
-- Table: t_mot
------------------------------------------------------------*/
CREATE TABLE t_mot(
	id_mot   INT IDENTITY (1,1) NOT NULL ,
	motMot   VARCHAR (25) NOT NULL UNIQUE,
	fktheme INT  NOT NULL ,
	CONSTRAINT prk_constraint_t_mot PRIMARY KEY NONCLUSTERED (id_mot)
);


/*------------------------------------------------------------
-- Table: t_theme
------------------------------------------------------------*/
CREATE TABLE t_theme(
	id_theme INT IDENTITY (1,1) NOT NULL ,
	theTheme VARCHAR (25) NOT NULL UNIQUE,
	CONSTRAINT prk_constraint_t_theme PRIMARY KEY NONCLUSTERED (id_theme)
);



ALTER TABLE t_image ADD CONSTRAINT FK_t_image_id_mot FOREIGN KEY (fkmot) REFERENCES t_mot(id_mot);
ALTER TABLE t_mot ADD CONSTRAINT FK_t_mot_id_theme FOREIGN KEY (fktheme) REFERENCES t_theme(id_theme);
